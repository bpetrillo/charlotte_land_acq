# LandTrack Environment Variables
# Copy this file to .env and fill in your API keys.

# Free tier: 1,000 requests/month. Sign up at https://regrid.com/api

# Property intelligence including ownership, sales history, and vacant land.
# Free trial available at https://api.attomdata.com
# VITE_ATTOM_API_KEY=your_attom_key_here
